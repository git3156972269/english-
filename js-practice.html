1. 变量定义 & 变量类型检测

核心知识点：let/const 声明、typeof 类型检测、基本数据类型（字符串/数字/布尔/undefined/null）

// 1.1 定义不同类型的变量
let userName = "Claude 作业提交"; // 字符串类型
let userAge = 22; // 数字类型（整数）
let userScore = 92.5; // 数字类型（浮点数）
let isPass = true; // 布尔类型
let emptyVal = null; // 空类型
let undefinedVal; // 未定义类型

// 1.2 输出变量类型到页面（修复typeof null的历史bug）
document.write("<h3>变量类型检测结果</h3>");
document.write("用户名：" + userName + " → 类型：" + typeof(userName) + "<br>");
document.write("年龄：" + userAge + " → 类型：" + typeof(userAge) + "<br>");
document.write("分数：" + userScore + " → 类型：" + typeof(userScore) + "<br>");
document.write("是否通过：" + isPass + " → 类型：" + typeof(isPass) + "<br>");
// 修复typeof null返回object的bug，改用=== null判断
document.write("空值：" + emptyVal + " → 类型：" + (emptyVal === null ? "null" : typeof(emptyVal)) + "<br>");
document.write("未定义：" + undefinedVal + " → 类型：" + typeof(undefinedVal) + "<br><hr>");

2. 表达式（算术/比较/逻辑运算）

核心知识点：算术运算符、比较运算符、逻辑运算符、表达式运算结果

// 2.1 算术表达式
let num1 = 20;
let num2 = 7;
let add = num1 + num2; // 加法
let sub = num1 - num2; // 减法
let mul = num1 * num2; // 乘法
let div = num1 / num2; // 除法
let mod = num1 % num2; // 取余

// 2.2 比较表达式
let isGreater = num1 > num2; // 大于
let isEqual = num1 === 20; // 全等（值+类型）

// 2.3 逻辑表达式
let hasScore = userScore > 60 && userScore <= 100; // 逻辑与
let isActive = userAge < 18 || userAge > 60; // 逻辑或

// 2.4 输出表达式结果
document.write("<h3>表达式运算结果</h3>");
document.write(num1 + " + " + num2 + " = " + add + "<br>");
document.write(num1 + " > " + num2 + " → " + isGreater + "<br>");
document.write("分数合法（60-100）→ " + hasScore + "<br>");

3. 流程控制（if-else/switch/循环）

核心知识点：条件分支语句、循环语句

// 3.1 if-else 流程控制（成绩评级）
document.write("<h3>成绩评级（if-else 流程）</h3>");
if (userScore >= 90) {
    document.write("分数 " + userScore + " → 评级：优秀 <br>");
} else if (userScore >= 80) {
    document.write("分数 " + userScore + " → 评级：良好 <br>");
} else if (userScore >= 60) {
    document.write("分数 " + userScore + " → 评级：及格 <br>");
} else {
    document.write("分数 " + userScore + " → 评级：不及格 <br>");
}

// 3.2 switch 流程控制（月份季节）
document.write("<h3>季节判断（switch 流程）</h3>");
let month = 4;
switch(month) {
    case 3: case 4: case 5:
        document.write("月份 " + month + " → 春季 <br>");
        break;
    case 6: case 7: case 8:
        document.write("月份 " + month + " → 夏季 <br>");
        break;
    case 9: case 10: case 11:
        document.write("月份 " + month + " → 秋季 <br>");
        break;
    case 12: case 1: case 2:
        document.write("月份 " + month + " → 冬季 <br>");
        break;
    default:
        document.write("月份 " + month + " → 无效月份 <br>");
}

// 3.3 for 循环（输出1-5）
document.write("<h3>1-5 数字循环（for 循环）</h3>");
for (let i = 1; i <= 5; i++) {
    document.write("循环值：" + i + " <br>");
}

4. 函数定义与调用

核心知识点：函数声明、参数传递、返回值、函数调用

// 4.1 无返回值函数（打印信息）
function showUserInfo() {
    document.write("<h3>用户信息（无返回值函数）</h3>");
    document.write("姓名：" + userName + " <br>");
    document.write("年龄：" + userAge + " <br>");
}

// 4.2 有返回值函数（计算平均分）
function calcAverage(scores) {
    let sum = 0;
    for (let score of scores) {
        sum += score;
    }
    return sum / scores.length;
}

// 4.3 函数调用
showUserInfo(); // 调用无参函数
let scoreArr = [92, 85, 96];
let avgScore = calcAverage(scoreArr); // 调用有参函数
document.write("<h3>平均分计算（有返回值函数）</h3>");
document.write("成绩数组：" + scoreArr + " → 平均分：" + avgScore.toFixed(2) + "<br>");
